java import java.awt.datatransfer.Clipboard; import java.awt.datatransfer.DataFlavor; import java.awt.datatransfer.StringSelection; import java.awt.datatransfer.Transferable; public class CopyTextWithFBPixels { public static void main(String[] args) { String text = "Texto de exemplo para copiar."; String svg = ""; String pixel = ""; String copiedText = copyTextWithFBPixels(text, svg, pixel); System.out.println("Texto copiado: " + copiedText); } public static String copyTextWithFBPixels(String text, String svg, String pixel) { StringBuilder stringBuilder = new StringBuilder(); stringBuilder.append(text); stringBuilder.append(svg); stringBuilder.append(pixel); String copiedText = stringBuilder.toString(); Clipboard clipboard = java.awt.Toolkit.getDefaultToolkit().getSystemClipboard(); Transferable transferable = new StringSelection(copiedText); clipboard.setContents(transferable, null); return copiedText; } }
